# Installation:
       For Linux Users:
       a) git clone https://github.com/machine1337/iplocation
       b) chmod +x run.sh
       c) ./run.sh
       
       For Termux Users:
       a) pkg update && pkg upgrade
       b) git clone https://github.com/machine1337/iplocation
       c) chmod +x run.sh
       d) ./run.sh
       
# Pre-requisite:
       a) curl
       b) git
       c) jq
